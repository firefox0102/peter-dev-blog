import { SpeakingCard } from "../components/SpeakingCard"

# public-speaking

Here's a collection of some conference talks I've given!

<SourceList filter="speaking">
  {(writing) => (
    <Flex
      sx={{
        flexWrap: "wrap",
        ml: (theme) => `-${theme.space[2]}px`,
        mr: (theme) => `-${theme.space[2]}px`,
      }}
    >
      {writing
        .filter((edge) => !edge.node.frontmatter.isPrivate)
        .map((edge, index) => {
          const {
            frontmatter: { title, featuredImage, tags, date, author },
            excerpt,
          } = edge.node
          return (
            <SpeakingCard
              key={index}
              title={title}
              featuredImage={featuredImage}
              tags={tags}
              date={date}
              excerpt={excerpt}
              link={author}
            />
          )
        })}
    </Flex>
)}

</SourceList>
